import React, { useState, useEffect } from 'react';
import axios from 'axios';
import { AnimatePresence, motion } from 'framer-motion';
import Swal from 'sweetalert2';
import WeatherCard from './components/WeatherCard';
import SearchBar from './components/SearchBar';
import LocationBtn from './components/LocationBtn';
import styles from './App.module.css';

const API_KEY = '8d86be893d2f42a188d131751260905'; // Replace with your key
const BASE_URL = 'https://api.weatherapi.com/v1';

const App = () => {
  const [weather, setWeather] = useState(null);
  const [loading, setLoading] = useState(false);
  const [unit, setUnit] = useState('C');
  const [bgClass, setBgClass] = useState('default');

  const fetchWeather = async (city) => {
    setLoading(true);
    try {
      const res = await axios.get(`${BASE_URL}/current.json`, {
        params: { key: API_KEY, q: city, aqi: 'yes' }
      });
      const data = res.data;
      const weatherData = {
        name: data.location.name,
        country: data.location.country,
        local_time: data.location.localtime,
        temp: data.current.temp_c,
        feels_like: data.current.feelslike_c,
        humidity: data.current.humidity,
        wind_speed: data.current.wind_kph,
        uv: data.current.uv,
        aqi: data.current.air_quality?.['us-epa-index'] || 'N/A',
        condition: data.current.condition.text,
        icon: data.current.condition.icon,
        isDay: data.current.is_day === 1,
      };
      setWeather(weatherData);
      setBgClass(getBgClass(weatherData.condition, weatherData.isDay));
    } catch (err) {
      Swal.fire({
        icon: 'error',
        title: 'City not found',
        text: 'Please try again with a valid city name.',
        background: '#1e293b',
        color: '#f1f5f9',
        confirmButtonColor: '#60a5fa',
      });
    } finally {
      setLoading(false);
    }
  };

  const fetchByLocation = () => {
    if (!navigator.geolocation) {
      Swal.fire('Error', 'Geolocation not supported', 'error');
      return;
    }
    navigator.geolocation.getCurrentPosition(
      async (pos) => {
        const { latitude, longitude } = pos.coords;
        setLoading(true);
        try {
          const res = await axios.get(`${BASE_URL}/current.json`, {
            params: { key: API_KEY, q: `${latitude},${longitude}`, aqi: 'yes' }
          });
          const data = res.data;
          const weatherData = {
            name: data.location.name,
            country: data.location.country,
            local_time: data.location.localtime,
            temp: data.current.temp_c,
            feels_like: data.current.feelslike_c,
            humidity: data.current.humidity,
            wind_speed: data.current.wind_kph,
            uv: data.current.uv,
            aqi: data.current.air_quality?.['us-epa-index'] || 'N/A',
            condition: data.current.condition.text,
            icon: data.current.condition.icon,
            isDay: data.current.is_day === 1,
          };
          setWeather(weatherData);
          setBgClass(getBgClass(weatherData.condition, weatherData.isDay));
        } catch (err) {
          Swal.fire('Error', 'Could not fetch location weather', 'error');
        } finally {
          setLoading(false);
        }
      },
      () => {
        Swal.fire('Location denied', 'Please allow location access', 'warning');
      }
    );
  };

  const getBgClass = (condition, isDay) => {
    const cond = condition.toLowerCase();
    if (cond.includes('sunny') || cond.includes('clear')) return isDay ? 'sunny' : 'night';
    if (cond.includes('cloud')) return 'cloudy';
    if (cond.includes('rain') || cond.includes('drizzle')) return 'rainy';
    if (cond.includes('snow')) return 'snowy';
    if (cond.includes('fog') || cond.includes('mist')) return 'foggy';
    return 'default';
  };

  useEffect(() => {
    fetchWeather('London');
  }, []);

  return (
    <div className={`${styles.app} ${styles[bgClass]}`}>
      <div className={styles.container}>
        <div className={styles.topBar}>
          <SearchBar onSearch={fetchWeather} loading={loading} />
          <LocationBtn onLocation={fetchByLocation} loading={loading} />
        </div>

        <AnimatePresence mode="wait">
          {loading ? (
            <motion.div 
              className={styles.loaderWrapper}
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              exit={{ opacity: 0 }}
            >
              <div className={styles.spinner}></div>
            </motion.div>
          ) : (
            weather && <WeatherCard weather={weather} unit={unit} setUnit={setUnit} />
          )}
        </AnimatePresence>

        <footer className={styles.footer}>
          <p>Weather data by WeatherAPI.com</p>
        </footer>
      </div>
    </div>
  );
};

export default App;

3. components/WeatherCard.jsx
jsx
import React from 'react';
import { motion } from 'framer-motion';
import { 
  WiThermometer, WiHumidity, WiStrongWind, WiDaySunny,
  WiCelsius, WiFahrenheit
} from 'react-icons/wi';
import { FaMapMarkerAlt } from 'react-icons/fa';
import { BsFillSunFill, BsFillMoonFill } from 'react-icons/bs';
import styles from './WeatherCard.module.css';

const WeatherCard = ({ weather, unit, setUnit }) => {
  if (!weather) return null;

  const { 
    name, country, temp, feels_like, humidity, wind_speed, 
    uv, aqi, local_time, condition, icon, isDay 
  } = weather;

  const tempDisplay = unit === 'C' ? temp : (temp * 9/5 + 32);
  const feelsDisplay = unit === 'C' ? feels_like : (feels_like * 9/5 + 32);
  const windDisplay = unit === 'C' ? wind_speed : (wind_speed * 0.621371);
  const windUnit = unit === 'C' ? 'km/h' : 'mph';

  return (
    <motion.div 
      className={styles.card}
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.5 }}
    >
      <div className={styles.cardHeader}>
        <div>
          <h2 className={styles.cityName}>
            <FaMapMarkerAlt className={styles.iconMap} />
            {name}, {country}
          </h2>
          <p className={styles.localTime}>{local_time}</p>
        </div>
        <div className={styles.tempToggle}>
          <button 
            className={`${styles.tempBtn} ${unit === 'C' ? styles.active : ''}`}
            onClick={() => setUnit('C')}
          >
            <WiCelsius />
          </button>
          <button 
            className={`${styles.tempBtn} ${unit === 'F' ? styles.active : ''}`}
            onClick={() => setUnit('F')}
          >
            <WiFahrenheit />
          </button>
        </div>
      </div>

      <div className={styles.weatherMain}>
        <div className={styles.conditionIcon}>
          {isDay ? <BsFillSunFill className={styles.sunIcon} /> : <BsFillMoonFill className={styles.moonIcon} />}
          <img src={icon} alt={condition} className={styles.weatherIcon} />
        </div>
        <div className={styles.tempDisplay}>
          <span className={styles.tempValue}>{Math.round(tempDisplay)}</span>
          <span className={styles.tempUnit}>°{unit}</span>
          <p className={styles.conditionText}>{condition}</p>
        </div>
      </div>

      <div className={styles.detailsGrid}>
        <div className={styles.detailItem}>
          <WiThermometer className={styles.detailIcon} />
          <span>Feels like <strong>{Math.round(feelsDisplay)}°{unit}</strong></span>
        </div>
        <div className={styles.detailItem}>
          <WiHumidity className={styles.detailIcon} />
          <span>Humidity <strong>{humidity}%</strong></span>
        </div>
        <div className={styles.detailItem}>
          <WiStrongWind className={styles.detailIcon} />
          <span>Wind <strong>{Math.round(windDisplay)} {windUnit}</strong></span>
        </div>
        <div className={styles.detailItem}>
          <WiDaySunny className={styles.detailIcon} />
          <span>UV Index <strong>{uv}</strong></span>
        </div>
        <div className={styles.detailItem}>
          <i className="fas fa-leaf" style={{color: '#4ade80', marginRight: '6px'}}></i>
          <span>AQI <strong>{aqi}</strong></span>
        </div>
      </div>
    </motion.div>
  );
};

export default WeatherCard;

5. components/SearchBar.jsx
jsx
import React, { useState } from 'react';
import { motion } from 'framer-motion';
import Swal from 'sweetalert2';
import { FaSearch, FaSpinner } from 'react-icons/fa';
import styles from './SearchBar.module.css';

const SearchBar = ({ onSearch, loading }) => {
  const [city, setCity] = useState('');

  const handleSubmit = (e) => {
    e.preventDefault();
    if (city.trim() === '') {
      Swal.fire({
        icon: 'warning',
        title: 'Oops...',
        text: 'Please enter a city name!',
        background: '#1e293b',
        color: '#f1f5f9',
        confirmButtonColor: '#60a5fa',
      });
      return;
    }
    onSearch(city.trim());
    setCity('');
  };

  return (
    <motion.form 
      className={styles.searchForm}
      onSubmit={handleSubmit}
      initial={{ opacity: 0, y: -10 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.4 }}
    >
      <div className={styles.searchWrapper}>
        <FaSearch className={styles.searchIcon} />
        <input
          type="text"
          className={styles.searchInput}
          placeholder="Search city..."
          value={city}
          onChange={(e) => setCity(e.target.value)}
        />
        <button type="submit" className={styles.searchBtn} disabled={loading}>
          {loading ? <FaSpinner className={styles.spinner} /> : 'Search'}
        </button>
      </div>
    </motion.form>
  );
};

export default SearchBar;
